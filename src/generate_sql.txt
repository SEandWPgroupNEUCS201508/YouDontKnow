create database ydk;
use ydk;
create table user (
    id int primary key auto_increment,
    username varchar(20) not null,
    password varchar(100) not null,
    email varchar(100)
);
create table comment (
    id int primary key auto_increment,
    comment text,
    user_id int not null,
    article_id int not null,
    comment_id int,
    published_date date,
    published_time time
);
create table article (
    id int primary key auto_increment,
    title varchar(100) not null,
    content text,
    user_id int not null,
    forum varchar(20),
    published_date date,
    published_time time
);
show tables;
